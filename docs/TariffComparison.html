def to_s
  sc_str = "%5.2f" % sc
  tc_str = "%5.2f" % tc
  total_str = "£%5.2f" % total
  saving_str = "£%5.2f" % saving
  code_str = "%28s" % (comparator? ? "*** " + code : code)
  "#{code_str}: Standing charges: #{sc_str}, tariff charges #{tc_str}; total #{total_str}, saving: #{saving_str}"
end
